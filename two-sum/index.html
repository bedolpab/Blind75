TWO_SUM

ARRAY

Iterate through the array once, storing each number and its index in a set. At each step, check if the current element's complement (target − value) is already in the map. A match means both indices are found in O(n) time instead of O(n²).

RUST 

use std::collections::HashSet;

impl Solution {
    pub fn two_sum(nums: Vec, target: i32) -> Vec {
        let mut set: HashSet = HashSet::new();

        for (i, &n) in nums.iter().enumerate() {
            let compl = target - n;

            if set.contains(&compl) {
                let j = nums.iter().position(|&x| x == compl).unwrap();
                return vec![j as i32, i as i32];
            }

            set.insert(n);
        }

        vec![]

    }
}
